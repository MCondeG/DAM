Create database Rama;

use Rama

	create table EQUIPO(
		codigo varchar (1),
		nombre varchar (30),
		ciudad varchar (20),
		web varchar (100),
		puntos integer,
		primary key (codigo))
		engine = innoDB;
		
	create table JUGADOR(
		id_jugador varchar (3),
		nombre varchar (30),
		apellido varchar (30),
		puesto varchar (10),
		id_capitan varchar (3),
		fecha_alta date,
		salario float,
		num_equipo varchar (1),
		altura float,
		primary key (id_jugador),
		foreign key (num_equipo) references EQUIPO (codigo))
		engine = innoDB;
		
	alter table JUGADOR add
		foreign key (id_capitan) references JUGADOR (id_jugador);
		
	create table Partidos(
		elocal varchar (1),
		evisitante varchar (1),
		resultado varchar (10),
		fecha date,
		arbitro varchar (1),
		primary key (elocal, evisitante, fecha),
		foreign key (elocal) references EQUIPO (codigo),
		foreign key (evisitante) references EQUIPO (codigo))
		engine = innoDB;
