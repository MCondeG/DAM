Create database Latiendadeinformatica;

use Latiendadeinformatica

	create table FABRICANTES(
		codigo varchar (10),
		nombre varchar (20),
		primary key (codigo))
		engine=InnoDB;
	
	create table ARTICULOS(
		nombre varchar (20),
		precio float,
		fabricante varchar (10),
		primary key (nombre),
		index (fabricante),
		foreign key (fabricante) references FABRICANTES(codigo))
		engine=InnoDB;
