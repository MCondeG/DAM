Create database LosInvestigadores;

use LosInvestigadores

	create table FACULTAD(
		codigo varchar (10),
		nombre varchar (30),
		primary key (codigo))
		engine=InnoDB;
		
	create table EQUIPO(
		numeroSerie varchar (10),
		nombre varchar (30),
		facultad varchar (10),
		primary key (numeroSerie),
		foreign key (facultad) references FACULTAD(codigo))
		engine=InnoDB;
		
	create table INVESTIGADOR(
		DNI varchar (10),
		nomApell varchar (50),
		facultad varchar (10),
		primary key (DNI),
		foreign key (facultad) references FACULTAD(codigo))
		engine=InnoDB;
		
	create table Reserva(
		DNI varchar (10),
		numSerie varchar (10),
		comienzo date,
		fin date,
		primary key (DNI,numSerie,comienzo),
		foreign key (DNI) references INVESTIGADOR(DNI),
		foreign key (numSerie) references EQUIPO(numeroSerie))
		engine=InnoDB;
