Create database Empleados;

use Empleados

	create table DEPARTAMENTO(
		codigo varchar (10),
		nombre varchar (20),
		presupuesto float,
		primary key (codigo))
		engine=InnoDB;
	
	create table EMPLEADOS(
		DNI varchar (10),
		nombre varchar (20),
		apellido varchar (20),
		departamento varchar (10),
		primary key (DNI),
		index (departamento),
		foreign key (departamento) references DEPARTAMENTO(codigo))
		engine=InnoDB;
